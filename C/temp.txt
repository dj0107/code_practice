int compare(char *a,char *b) {
    int i = 0;
    int matched = 1;
    while (b[i]) {
        if (!a[i]) {
            matched = 0;
            break;
        }
        if (b[i] != a[i]) {
            matched = 0;
            break;
        }
        i++;
    }
    if (a[i]) {matched = 0;}
    return matched;

}
